IntersectionObserver
触发条件：
观察到元素，触发一次；
观察不到元素，触发一次；
观察元素：根据回调函数cb[callback]的isIntersecting判断是否进入，并执行相应的业务逻辑；；如果已经观察到了，就解除观察；
    let imgs = document.querySelectorAll('img')//获取dom
    let cb = (entries) => {
        entries.forEach(entry => {
            if (entry.isIntersecting) {//观察到元素，开始渲染
                let img = entry.target
                img.src = img.getAttribute('data-src')
                observer.unobserve(img)//渲染完毕后解除观察
            }
        })
    }
    let observer = new IntersectionObserver(cb)
    imgs.forEach(item => {//循环为每个元素绑定观察
        observer.observe(item)
    })
