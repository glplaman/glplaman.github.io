静态资源服务器

.服务器代理

server: {
  proxy: {
    "/api": {
      target: "http://localhost:3000",
      changeOrigin: true,
      rewrite: (path) => path.replace(/^\/api/, ""),
    },
  },
},

.使用原生 fetch 请求

fetch('/api/get')
  .then(res => res.json())
  .then(data => {
    console.log(data)
  }
)

.使用 axios 请求

axios.get('/api/get')
  .then(data => {
    console.log(data)
  })
